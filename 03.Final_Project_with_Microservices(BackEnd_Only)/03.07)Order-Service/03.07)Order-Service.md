### ðŸ’° Price calculation + stock validation

```js
        let priceAmount = 0; // final order total

        const orderItems = cartResponse.data.cart.items.map((item) => {

            // Find matching product fetched earlier
            const product = products.find(
                p => p._id === item.productId
            );

            // Prevent ordering more than available stock
            if (product.stock < item.quantity) {
                throw new Error(
                    `Product ${product.title} is out of stock or insufficient stock`
                );
            }

            // Total price for this item
            const itemTotal = product.price.amount * item.quantity;

            // Add to order total
            priceAmount += itemTotal;

            // Shape order item according to schema
            return {
                product: item.productId,
                quantity: item.quantity,
                price: {
                    amount: itemTotal,
                    currency: product.price.currency
                }
            };
        });
```

### Important concepts here

| Concept           | Why                       |
| ----------------- | ------------------------- |
| Stock check       | Prevent overselling       |
| Server-side price | Client price is untrusted |
| Calculated total  | Prevent tampering         |

## 3ï¸âƒ£ `getMyOrders` (Pagination Explained)

```js
async function getMyOrders(req, res) {

    const user = req.user;

    // Pagination parameters
    const page = parseInt(req.query.page) || 1;
    const limit = parseInt(req.query.limit) || 10;

    // How many documents to skip
    const skip = (page - 1) * limit;

    try {
        // Fetch paginated orders
        const orders = await orderModel
            .find({ user: user.id })
            .skip(skip)
            .limit(limit)
            .exec();

        // Total count for frontend pagination
        const totalOrders =
            await orderModel.countDocuments({ user: user.id });

        res.status(200).json({
            orders,
            meta: {
                total: totalOrders,
                page,
                limit
            }
        });
```

## 4ï¸âƒ£ `getOrderById` (Authorization logic)

```js
async function getOrderById(req, res) {

    const user = req.user;
    const orderId = req.params.id;

    try {
        const order = await orderModel.findById(orderId);

        // Order not found
        if (!order) {
            return res.status(404).json({
                message: "Order not found"
            });
        }

        // Ensure order belongs to user
        if (order.user.toString() !== user.id) {
            return res.status(403).json({
                message: "Forbidden: You do not have access to this order"
            });
        }

        res.status(200).json({ order });
```

## 5ï¸âƒ£ `cancelOrderById`

```js
async function cancelOrderById(req, res) {

    const user = req.user;
    const orderId = req.params.id;

    try {
        const order = await orderModel.findById(orderId);

        if (!order) {
            return res.status(404).json({ message: "Order not found" });
        }

        if (order.user.toString() !== user.id) {
            return res.status(403).json({ message: "Forbidden" });
        }

        // Business rule enforcement
        if (order.status !== "PENDING") {
            return res.status(409).json({
                message: "Order cannot be cancelled at this stage"
            });
        }

        order.status = "CANCELLED";
        await order.save();

        res.status(200).json({ order });
```

```js
async function updateOrderAddress(req, res) {

    const user = req.user;
    const orderId = req.params.id;

    try {
        const order = await orderModel.findById(orderId);

        if (!order) {
            return res.status(404).json({ message: "Order not found" });
        }

        if (order.user.toString() !== user.id) {
            return res.status(403).json({ message: "Forbidden" });
        }

        // Address can only be updated before processing
        if (order.status !== "PENDING") {
            return res.status(409).json({
                message: "Order address cannot be updated at this stage"
            });
        }

        // Replace address atomically
        order.shippingAddress = {
            street: req.body.shippingAddress.street,
            city: req.body.shippingAddress.city,
            state: req.body.shippingAddress.state,
            zip: req.body.shippingAddress.pincode,
            country: req.body.shippingAddress.country,
        };

        await order.save();

        res.status(200).json({ order });
```
